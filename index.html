Ticket System

Command to open private ticket channels with permissions.

JavaScript 
import { SlashCommandBuilder, PermissionFlagsBits, ChannelType } from 'discord.js';

export default {
  data: new SlashCommandBuilder()
    .setName('ticket')
    .setDescription('Open a private ticket channel')
    .addStringOption(option =>
      option.setName('reason').setDescription('Reason for the ticket').setRequired(true)
    ),

  async execute(interaction) {
    const reason = interaction.options.getString('reason');

    const ticketChannel = await interaction.guild.channels.create({
      name: `ticket-${interaction.user.username}`,
      type: ChannelType.GuildText,
      permissionOverwrites: [
        {
          id: interaction.guild.roles.everyone,
          deny: [PermissionFlagsBits.ViewChannel]
        },
        {
          id: interaction.user.id,
          allow: [PermissionFlagsBits.ViewChannel, PermissionFlagsBits.SendMessages]
        },
        {
          id: interaction.client.user.id,
          allow: [PermissionFlagsBits.ViewChannel, PermissionFlagsBits.SendMessages]
        }
      ]
    });

    await ticketChannel.send(
      `üé´ Ticket opened by <@${interaction.user.id}> for **${reason}**. A staff member will help you soon.`
    );

    await interaction.reply({ content: `‚úÖ Ticket created: ${ticketChannel}`, ephemeral: true });
  }
};

Auto-Moderation System

Deletes messages containing banned words and warns users.

JavaScript 
export default {
  name: 'messageCreate',
  once: false,

  async execute(message, client, config) {
    if (!message.guild || message.author.bot) return;

    const content = message.content.toLowerCase();
    const matchedWord = config.badWords.find(word => content.includes(word));
    if (!matchedWord) return;

    try {
      await message.delete();
      await message.channel.send({
        content: `‚ö†Ô∏è ${message.author}, please avoid using inappropriate language.`
      });

      console.log(`[AutoMod] Deleted message from ${message.author.tag} containing "${matchedWord}"`);

      const logChannel = message.guild.channels.cache.find(
        ch => ch.name === config.logChannel
      );
      if (logChannel) {
        await logChannel.send({
          content: `üõ°Ô∏è **AutoMod**: Deleted a message from ${message.author} in ${message.channel}\n> **Content**: ||${message.content}||\n> **Reason**: Contains banned word "${matchedWord}"`
        });
      }
    } catch (error) {
      console.error('AutoMod error:', error);
      await message.channel.send({
        content: `‚ö†Ô∏è Could not delete message due to permissions.`,
        ephemeral: true
      }).catch(() => {});
    }
  }
};

Welcome System

Sends customizable welcome messages to new members.

JavaScript 
export default {
  name: 'guildMemberAdd',
  once: false,

  async execute(member, client, config) {
    try {
      let channel = member.guild.systemChannel;
      if (!channel && config.welcomeChannel) {
        channel = member.guild.channels.cache.find(
          ch => ch.name === config.welcomeChannel && ch.isTextBased()
        );
      }

      if (!channel) {
        console.log(`[Welcome] No channel found to send welcome for ${member.user.tag}`);
        return;
      }

      const messageTemplate = config.welcomeMessage || "üëã Welcome, {user}!";
      const finalMessage = messageTemplate.replace('{user}', `<@${member.id}>`);

      await channel.send({ content: finalMessage });
      console.log(`[Welcome] Sent welcome message for ${member.user.tag} in #${channel.name}`);
    } catch (error) {
      console.error('[Welcome] Failed to send welcome message:', error);
    }
  }
};
